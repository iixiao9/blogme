Golang将int类型转换为string类型

2021.05.13
#Golang 
package main

import (
	"fmt"
	"strconv"
)

func main() {
	var a int = 65
	b := strconv.Itoa(a)
	fmt.Println(b)
	a, _ = strconv.Atoi(b)
	fmt.Println(a)
}

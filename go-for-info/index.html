Golang中for语句详解

2021.05.13
#Golang

for循环第一种

func main() {
	var num = 1
	for {
		if num > 5 {
			break
		}
        num++
		fmt.Println(num)
	}
	fmt.Println("程序结束!")
}

func main() {
	var num = 1
	for i := 0; i <= 3; i++ {
